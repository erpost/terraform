# Commands to register GKE with Anthos
# Bug is being tracked: https://issuetracker.google.com/issues/184423917

gcloud --project <project_id> container clusters list --uri
gcloud --project <project_id> container hub memberships register <membership name (use cluster name)> \
--gke-uri=<uri of gke cluster> \
--service-account-key-file=<service account json key>
